class Pattern19
{
public static void  main(String args[])
	{
	  int n = 5; // size
	  int i;
	  int j;
	  
	  for (i = n - 1; i >= 0; i--) 
	  {
		/*
	   outer gap loop
		*/
	  for (j = n - 1; j > i; j--) 
	  {
	    System.out.print(" ">);
	}
	    System.out.printf("%c", i + 65); // 65-ASCII of 'A'
		/*
		* inner gap loop
		*/
		
		for (j = 1; j < (i * 2); j++) {
		System.out.print(" ");
		}
		if (i >= 1) 
		{
		System.out.printf("%c", i + 65); // 65-ASCII of 'A'
		}
		System.out.print("\n");
	}
   }
}
