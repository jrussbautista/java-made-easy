class Pattern13
{
public static void  main(String args[])
	{
	 int max_stars=7; //change the value of it to increase or decrease the size of piramid
	 //take odd no. for odd no. of stars in each row 1-3-5 etc
	 //take even no. for even no. stars in each row, 2-4-6 etc
	  int p_space=max_stars/2;
	  int p_height=max_stars-p_space;
	   for( int i =p_height;i>=1;i --)
	  {
	    for( int j =p_space;j>=i;j --)
	   {
	     System.out.print( " ");
		}
		  for( int k =1;k<=max_stars;k ++)
		 {
		   System.out.print( "*");
		 }
		 max_stars-=2;
		  System.out.println();
		 }
	}
}
