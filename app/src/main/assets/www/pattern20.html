class Pattern20
{
public static void  main(String args[])
	 int n = 5; //size
	 int i;
	 int j;
	 for (i = n;i >= 1;i--)
	 {
	  for (j = n;j >= 1;j--)
	  {
	   if (i == j || i + j == n + 1)
	   {
	    System.out.print(i);
		System.out.print(" ");
	    }
		 else
		 {
		   System.out.print("  ");
		 }
		}
		   System.out.print("\n");
	   }
	}
}
