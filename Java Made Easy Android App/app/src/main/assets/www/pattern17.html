class Pattern17
{
public static void  main(String args[])
	{
	 int size = 5;
	  /*
	 change value of 'size' to increase or decrease the size of pyramid
	  */
	  int s,i,j;
	  int space = size;
	  
	  for (s = 1;s <= size;s++)
	  {
	  
	    for (i = 1; i <= s; i++)
		{
		 System.out.print("*");
		}
		
		for (j = space; j > 1; j--)
		{
		 System.out.print("  ");
		}
		 for (i = ; i >= 1; i--)
		 {
		  System.out.print("*");
		 }
		  System.out.println("");
		  space--;
		 }
	}
}
