class Pattern15
{
public static void  main(String args[])
	{
	 int i,j,k;
	 int p_height = 5; // change it to increase or decrease pyramid height
	 
	 /*Top pyramid*/
	 
	 for (i = 1; i <= p_height; i++)
	 {
	  for (k = p_height - 1; k >= i; k--)
	  {
	   System.out.print(" ");
	  }
	   for (j = 1; j <= i; j++)
	   {
	    System.out.print("* ");
		}
		System.out.println("");
		}
		/*Bottom pyramid*/
		for (i = p_height - 1; i >= 1; i--)
		{
		 for (k = p_height - 1; k >= i; k--)
		 {
		  System.out.print(" ");
		  }
		   for (j = i; j >= 1; j--)
		   {
		    System.out.print("* ");
			}
			 System.out.println("");
		 }
	}
}
