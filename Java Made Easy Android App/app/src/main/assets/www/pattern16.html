class Pattern16
{
public static void  main(String args[])
	{
	 int size;
	 int i;
	 int j;
	 int s = 1;
	 
	 //change value of size to increase or decrease the size of pattern

	 for (size = 5;size >= 1;size--)
	 {
	  for (i = 1; i <= size; i++)
	  {
	    System.out.print("*");
	   }
	    for (j = s; j > 1; j--)
		{ 
		  System.out.print("  ");// 2 spaces
		 }
		   for (i = size; i >= 1; i--)
		   {
		    System.out.print("*");
		}
		    System.out.print("\n");
			s++;
		}
	}
}
